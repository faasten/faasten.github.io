  9def recvall(sock, n):
 10    # Helper function to recv n bytes or return None if EOF is hit
 11    data = bytearray()
 12    while len(data) < n:
 13        packet = sock.recv(n - len(data))
 14        if not packet:
 15            return None
 16        data.extend(packet)
 17    return bytes(data)

 28def convert_path(path):
 29    converted = []
 30    for comp in path:
 31        m = syscalls_pb2.PathComponent()

327    def create_blob(self, size: int = None):
328        """Create a nameless data object.
329
330        The implementation uses the content-addressed blob store.
331
332        Yield:
333            An instance of class NewBlob
334        """
335        req = syscalls_pb2.Syscall(createBlob=syscalls_pb2.BlobCreate(size=size))
336        self._send(req)
337        response = self._recv(syscalls_pb2.BlobResponse())
338        if response.success:
339            fd = response.fd
340            yield NewBlob(fd, self)
341            syscalls_pb2.Syscall(closeBlob=syscalls_pb2.BlobClose(fd=fd))
342            self._send(req)
343            response = self._recv(syscalls_pb2.BlobResponse())
344        else:
345            raise CreateBlobError

394class Blob():
395    def __init__(self, fd, syscall):
396        self.fd = fd
397        self.offset = 0
398        self.syscall = syscall

400    def _blob_read(self, offset=None, length=None):
401        req = syscalls_pb2.Syscall(readBlob=syscalls_pb2.BlobRead(fd=self.fd, offset=offset, length=length))
402        self.syscall._send(req)
403        response = self.syscall._recv(syscalls_pb2.BlobResponse())
404        if response.success:
405            return response.data
406        raise ReadBlobError
407
408    def read(self, size=None):
409        # if size is unspecified, implementation-dependent
410        # faasten now returns at most one block (4K) data
411        if size is None:
412            return self._blob_read()
413        else:
414            buf = bytearray(b'')
415            while size > 0:
416                data = self._blob_read(self.offset, size)
417                # reaches EOF
418                if len(data) == 0:
419                    return buf
420                buf.extend(data)
421                self.offset += len(data)
422                size -= len(data)
423            return buf

425    def tell(self):
426        return self.offset

428    def seek(self, offset, whence=0):
429        # TODO handle whence of 2: seek from end of file
430        if whence == 1:
431            self.offset += offset
432        else:
433            self.offset = offset

327    @contextmanager
328    def create_blob(self, size: int = None):
329        """Create a nameless data object.
330
331        The implementation uses the content-addressed blob store.
332
333        Yield:
334            An instance of class NewBlob
335        """
336        req = syscalls_pb2.Syscall(createBlob=syscalls_pb2.BlobCreate(size=size))
337        self._send(req)
338        response = self._recv(syscalls_pb2.BlobResponse())
339        if response.success:
340            fd = response.fd
341            yield NewBlob(fd, self)
342            syscalls_pb2.Syscall(closeBlob=syscalls_pb2.BlobClose(fd=fd))
343            self._send(req)
344            response = self._recv(syscalls_pb2.BlobResponse())
345        else:
346            raise CreateBlobError
